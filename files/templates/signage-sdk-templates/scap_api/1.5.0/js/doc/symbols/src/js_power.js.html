 78                         platformInfoObj = {
 79                             webOSVer : 3,
 80                             chipset : result.boardType.split("_")[0]
 81                         };
 82                     } else {
 83                         platformInfoObj = {
 84                             webOSVer : 0,
 85                             chipset : ""
 86                         };
 87                     }

 90                     delete result.returnValue;
